#version 330

in  vec4  vPosition;
in  vec4  vColor;
out vec4  outColor;

uniform float time = 0.0;

void main ()
{ 
	outColor = vColor;
	outColor.r = (cos(time + vPosition.x * 10) + 1.0)/ 2.0;
	outColor.g = (cos(time + vPosition.y * 10) + 1.0)/ 2.0;
	outColor.b = (cos(time + vPosition.x * 10) + 1.0)/ 2.0;
}
